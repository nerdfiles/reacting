 * @summary a task-based interface.
 * @name act
 * @function
 * @version 0.0.0-novel.0
 * @since 0.0.0-novel.0
 * @param {} occasion.
 * @param {} action.
 * @returns Array .
 */
import {
  POST_KEY,
  OPTIONS_KEY
} from './form-actions'

const act = (occasion = [], action) => {
  switch (action.type) {
    case OPTIONS_KEY:
      return {
        next: ''
      }
    case POST_KEY:
      return {
        next: ''
      }
    default:
      break
  }
  return occasion
}

export default act
